// === 日報 Journal API (Google Apps Script) === // スプレッドシートを見やすい列構成で保存 + JSON APIとして読み書き const SS = SpreadsheetApp.getActiveSpreadsheet(); // 日付キーを YYYY-MM-DD 形式に正規化 function normalizeKey(val) { if (val instanceof Date) { const y = val.getFullYear(); const m = String(val.getMonth() + 1).padStart(2, '0'); const d = String(val.getDate()).padStart(2, '0'); return y + '-' + m + '-' + d; } return String(val); } // camelCase変換ヘルパー function toCamel(jsKey) { const map = { DAILY:'daily', WEEKLY:'weekly', MONTHLY:'monthly', MEMOS:'memos', WEEKLY_GOALS:'weeklyGoals', MONTHLY_GOALS:'monthlyGoals' }; return map[jsKey] || jsKey; } // ===== シート定義 ===== const SHEETS = { DAILY: 'daily', WEEKLY: 'weekly', MONTHLY: 'monthly', MEMOS: 'memos', WEEKLY_GOALS: 'weeklyGoals', MONTHLY_GOALS: 'monthlyGoals' }; // ===== 列定義（スプレッドシートのヘッダー） ===== const DAILY_COLS = ['日付','就寝','起床','体調','運動','運動メモ','モチベーション','モチベメモ','楽しかったこと','なぜ楽しかった','不愉快だったこと','なぜ不愉快','失敗','苦手','改善','FB','印象的な出来事','更新日']; const WEEKLY_COLS = ['週キー','達成度','良かったこと','改善できそう','来週頑張りたい','来週の目標案','更新日']; const MONTHLY_COLS = ['月','体重','達成度','良かったこと','改善できそう','楽しかったこと','来月の目標案','更新日']; const MEMO_COLS = ['ID','内容','登録日','解決済み']; const GOAL_COLS = ['キー','目標','更新日']; // ===== 初期化（常にヘッダーを最新に更新） ===== function initSheets() { const headers = { [SHEETS.DAILY]: DAILY_COLS, [SHEETS.WEEKLY]: WEEKLY_COLS, [SHEETS.MONTHLY]: MONTHLY_COLS, [SHEETS.MEMOS]: MEMO_COLS, [SHEETS.WEEKLY_GOALS]: GOAL_COLS, [SHEETS.MONTHLY_GOALS]: GOAL_COLS }; Object.entries(headers).forEach(([name, cols]) => { let s = SS.getSheetByName(name); if (!s) s = SS.insertSheet(name); // 常にヘッダーを設定（既存でも上書き） s.getRange(1, 1, 1, cols.length).setValues([cols]).setFontWeight('bold').setBackground('#e8eaf6'); }); } // ===== マイグレーション: 旧JSON形式 → 新しい列形式に変換 ===== // ★ Apps Scriptエディタで1回だけ手動実行してください ★ function migrateToColumns() { initSheets(); const log = []; // --- Daily --- const ds = SS.getSheetByName(SHEETS.DAILY); const dsLastRow = ds.getLastRow(); if (dsLastRow > 1) { // 全列読み込み（旧形式は2列目にJSON、新形式は多数列） const allRows = ds.getRange(2, 1, dsLastRow - 1, Math.max(ds.getLastColumn(), 3)).getValues(); // 旧JSON形式かどうか: B列(index 1)が {で始まる文字列 const oldRows = allRows.filter(r => { const b = String(r[1] || '').trim(); return b.startsWith('{') || b.startsWith('['); }); log.push('Daily: 旧形式行数=' + oldRows.length + ' / 全行数=' + allRows.length); if (oldRows.length > 0) { // 全データ行を削除してから再書き込み ds.deleteRows(2, dsLastRow - 1); let successCount = 0; allRows.forEach(r => { const key = normalizeKey(r[0]); if (!key) return; const b = String(r[1] || '').trim(); if (b.startsWith('{') || b.startsWith('[')) { try { const e = JSON.parse(r[1]); const newRow = ds.getLastRow() + 1; ds.appendRow(dailyToRow(key, e)); writeKeyCell(ds, newRow, 1, key); successCount++; } catch(ex) { log.push('Daily JSON parse error key=' + key + ': ' + ex.message); } } else { // 既に列形式のデータ → そのまま再書き込み const newRow = ds.getLastRow() + 1; ds.appendRow(r.slice(0, DAILY_COLS.length)); writeKeyCell(ds, newRow, 1, key); successCount++; } }); log.push('Daily: ' + successCount + '件変換完了'); } } else { log.push('Daily: データなし'); } // --- Weekly --- const ws = SS.getSheetByName(SHEETS.WEEKLY); const wsLastRow = ws.getLastRow(); if (wsLastRow > 1) { const allRows = ws.getRange(2, 1, wsLastRow - 1, Math.max(ws.getLastColumn(), 3)).getValues(); const oldRows = allRows.filter(r => String(r[1]||'').trim().startsWith('{')); log.push('Weekly: 旧形式行数=' + oldRows.length); if (oldRows.length > 0) { ws.deleteRows(2, wsLastRow - 1); let cnt = 0; allRows.forEach(r => { const key = normalizeKey(r[0]); if (!key) return; const b = String(r[1]||'').trim(); if (b.startsWith('{')) { try { const e = JSON.parse(r[1]); const newRow = ws.getLastRow() + 1; ws.appendRow(weeklyToRow(key, e)); writeKeyCell(ws, newRow, 1, key); cnt++; } catch(ex) { log.push('Weekly error key=' + key + ': ' + ex.message); } } else { ws.appendRow(r.slice(0, WEEKLY_COLS.length)); writeKeyCell(ws, ws.getLastRow(), 1, key); cnt++; } }); log.push('Weekly: ' + cnt + '件変換完了'); } } // --- Monthly --- const ms = SS.getSheetByName(SHEETS.MONTHLY); const msLastRow = ms.getLastRow(); if (msLastRow > 1) { const allRows = ms.getRange(2, 1, msLastRow - 1, Math.max(ms.getLastColumn(), 3)).getValues(); const oldRows = allRows.filter(r => String(r[1]||'').trim().startsWith('{')); log.push('Monthly: 旧形式行数=' + oldRows.length); if (oldRows.length > 0) { ms.deleteRows(2, msLastRow - 1); let cnt = 0; allRows.forEach(r => { const key = normalizeKey(r[0]); if (!key) return; const b = String(r[1]||'').trim(); if (b.startsWith('{')) { try { const e = JSON.parse(r[1]); const newRow = ms.getLastRow() + 1; ms.appendRow(monthlyToRow(key, e)); writeKeyCell(ms, newRow, 1, key); cnt++; } catch(ex) { log.push('Monthly error key=' + key + ': ' + ex.message); } } else { ms.appendRow(r.slice(0, MONTHLY_COLS.length)); writeKeyCell(ms, ms.getLastRow(), 1, key); cnt++; } }); log.push('Monthly: ' + cnt + '件変換完了'); } } // --- Memos --- const mms = SS.getSheetByName(SHEETS.MEMOS); const mmsLastRow = mms.getLastRow(); if (mmsLastRow > 1) { const allRows = mms.getRange(2, 1, mmsLastRow - 1, Math.max(mms.getLastColumn(), 3)).getValues(); const oldRows = allRows.filter(r => String(r[1]||'').trim().startsWith('{')); log.push('Memos: 旧形式行数=' + oldRows.length); if (oldRows.length > 0) { mms.deleteRows(2, mmsLastRow - 1); let cnt = 0; allRows.forEach(r => { const b = String(r[1]||'').trim(); if (b.startsWith('{')) { try { const memo = JSON.parse(r[1]); mms.appendRow(memoToRow(memo)); cnt++; } catch(ex) { log.push('Memo error: ' + ex.message); } } else { mms.appendRow(r.slice(0, MEMO_COLS.length)); cnt++; } }); log.push('Memos: ' + cnt + '件変換完了'); } } // --- Goals --- [SHEETS.WEEKLY_GOALS, SHEETS.MONTHLY_GOALS].forEach(sn => { const gs = SS.getSheetByName(sn); const gsLastRow = gs.getLastRow(); if (gsLastRow > 1) { const oldRows = gs.getRange(2, 1, gsLastRow - 1, Math.max(gs.getLastColumn(), 3)).getValues(); gs.deleteRows(2, gsLastRow - 1); let cnt = 0; oldRows.forEach(r => { const key = normalizeKey(r[0]); if (!key) return; let val = r[1]; try { val = JSON.parse(val); } catch(ex) {} const newRow = gs.getLastRow() + 1; gs.appendRow(goalToRow(key, val)); writeKeyCell(gs, newRow, 1, key); cnt++; }); log.push(sn + ': ' + cnt + '件処理完了'); } }); const summary = log.join('\n'); Logger.log(summary); SpreadsheetApp.getUi().alert('マイグレーション完了！\n\n' + summary); } // ===== Daily: オブジェクト → 行 ===== function dailyToRow(key, e) { const failText = (e.failures || []).map(f => f.what + (f.why ? '（原因:' + f.why + '）' : '') + (f.lesson ? '→学び:' + f.lesson : '')).join('\n'); const weakText = (Array.isArray(e.weaknesses) ? e.weaknesses : (e.weakness ? [{content:e.weakness}] : [])).map(w => w.content).filter(Boolean).join('\n'); const impText = (Array.isArray(e.improvements) ? e.improvements : (e.improvement ? [{content:e.improvement}] : [])).map(w => w.content).filter(Boolean).join('\n'); const fbText = (e.feedbacks || []).map(f => '[' + (f.who||'?') + '] ' + (f.content||'')).join('\n'); // 楽しかったこと: 新形式 enjoyments[]={what,why} / 旧形式 enjoyWhat/enjoyWhy let enjoyWhatText = '', enjoyWhyText = ''; if (Array.isArray(e.enjoyments) && e.enjoyments.length > 0) { enjoyWhatText = e.enjoyments.map(en => en.what||'').join('\n'); enjoyWhyText = e.enjoyments.map(en => en.why||'').join('\n'); } else { enjoyWhatText = e.enjoyWhat || ''; enjoyWhyText = e.enjoyWhy || ''; } // 印象的な出来事: 新形式 impressions[]={content} / 旧形式 timeEntries[]={activity,hours} let impressionText = ''; if (Array.isArray(e.impressions) && e.impressions.length > 0) { impressionText = e.impressions.map(im => im.content||'').filter(Boolean).join('\n'); } else if (Array.isArray(e.timeEntries) && e.timeEntries.length > 0) { // 旧時間記録データは「活動:時間h」形式で移行 impressionText = e.timeEntries.map(t => t.activity + ':' + t.hours + 'h').join('\n'); } const exLabel = e.exercise === true ? 'はい' : e.exercise === false ? 'いいえ' : ''; const motivLabel = e.motivation ? ['','😫最低','😔低い','😐普通','😊高い','🔥最高'][e.motivation] || '' : ''; return [key, e.bedtime||'', e.wakeup||'', e.health||'', exLabel, e.exMemo||'', motivLabel, e.motivMemo||'', enjoyWhatText, enjoyWhyText, e.badWhat||'', e.badWhy||'', failText, weakText, impText, fbText, impressionText, e.savedAt || new Date().toISOString()]; } // ===== Daily: 行 → オブジェクト（API読み込み用） ===== function rowToDaily(row) { const exMap = {'はい':true, 'いいえ':false}; const motivMap = {'😫最低':1,'😔低い':2,'😐普通':3,'😊高い':4,'🔥最高':5}; const e = {}; e.bedtime = row[1] || ''; e.wakeup = row[2] || ''; e.health = row[3] || ''; e.exercise = exMap[row[4]] !== undefined ? exMap[row[4]] : null; e.exMemo = row[5] || ''; e.motivation = motivMap[row[6]] || null; e.motivMemo = row[7] || ''; // 楽しかったこと: 複数行 → enjoyments[]={what,why} const whatLines = String(row[8]||'').split('\n').filter(Boolean); const whyLines = String(row[9]||'').split('\n'); if (whatLines.length > 0) { e.enjoyments = whatLines.map((w, i) => ({what: w, why: whyLines[i]||''})); } else { e.enjoyments = []; } e.badWhat = row[10] || ''; e.badWhy = row[11] || ''; // 失敗: テキストから復元 const failStr = String(row[12]||''); e.failures = failStr ? failStr.split('\n').filter(Boolean).map(line => { const m1 = line.match(/^(.+?)(?:（原因:(.+?)）)?(?:→学び:(.+))?$/); return m1 ? {what:m1[1].trim(), why:(m1[2]||'').trim(), lesson:(m1[3]||'').trim()} : {what:line, why:'', lesson:''}; }) : []; // 苦手 const weakStr = String(row[13]||''); e.weaknesses = weakStr ? weakStr.split('\n').filter(Boolean).map(s => ({content:s})) : []; // 改善 const impStr = String(row[14]||''); e.improvements = impStr ? impStr.split('\n').filter(Boolean).map(s => ({content:s})) : []; // FB const fbStr = String(row[15]||''); e.feedbacks = fbStr ? fbStr.split('\n').filter(Boolean).map(line => { const m = line.match(/^\[(.+?)\]\s*(.*)$/); return m ? {who:m[1], content:m[2]} : {who:'?', content:line}; }) : []; // 印象的な出来事: 複数行 → impressions[]={content} const imprStr = String(row[16]||''); e.impressions = imprStr ? imprStr.split('\n').filter(Boolean).map(s => ({content:s})) : []; e.savedAt = row[17] || ''; return e; } // ===== Weekly: オブジェクト → 行 ===== function weeklyToRow(key, e) { return [key, e.achievement||'', e.good||'', e.improve||'', e.hard||'', e.nextGoals||'', e.savedAt || new Date().toISOString()]; } function rowToWeekly(row) { return { achievement:row[1]||'', good:row[2]||'', improve:row[3]||'', hard:row[4]||'', nextGoals:row[5]||'', savedAt:row[6]||'' }; } // ===== Monthly: オブジェクト → 行 ===== function monthlyToRow(key, e) { return [key, e.weight||'', e.achievement||'', e.good||'', e.improve||'', e.enjoyment||'', e.nextGoals||'', e.savedAt || new Date().toISOString()]; } function rowToMonthly(row) { return { weight:row[1]||'', achievement:row[2]||'', good:row[3]||'', improve:row[4]||'', enjoyment:row[5]||'', nextGoals:row[6]||'', savedAt:row[7]||'' }; } // ===== Memos: オブジェクト → 行 ===== function memoToRow(memo) { return [memo.id||'', memo.content||'', memo.date||'', memo.resolved ? 'はい' : 'いいえ']; } function rowToMemo(row) { return { id:row[0]||0, content:row[1]||'', date:row[2]||'', resolved:row[3]==='はい' }; } // ===== Goals: 行 → 値 ===== function goalToRow(key, val) { return [key, val||'', new Date().toISOString()]; } function rowToGoalObj(rows) { const obj = {}; rows.forEach(r => { obj[normalizeKey(r[0])] = r[1] || ''; }); return obj; } // ===== テキスト形式でキーを書き込むヘルパー ===== function writeKeyCell(sheet, row, col, key) { sheet.getRange(row, col).setNumberFormat('@').setValue(key); } // ===== GET: 全データを返す ===== function doGet(e) { initSheets(); const result = { daily:{}, weekly:{}, monthly:{}, memos:[], weeklyGoals:{}, monthlyGoals:{} }; // Daily const ds = SS.getSheetByName(SHEETS.DAILY); if (ds.getLastRow() > 1) { const rows = ds.getRange(2, 1, ds.getLastRow() - 1, DAILY_COLS.length).getValues(); rows.forEach(r => { const key = normalizeKey(r[0]); if (key) result.daily[key] = rowToDaily(r); }); } // Weekly const ws = SS.getSheetByName(SHEETS.WEEKLY); if (ws.getLastRow() > 1) { const rows = ws.getRange(2, 1, ws.getLastRow() - 1, WEEKLY_COLS.length).getValues(); rows.forEach(r => { const key = normalizeKey(r[0]); if (key) result.weekly[key] = rowToWeekly(r); }); } // Monthly const ms = SS.getSheetByName(SHEETS.MONTHLY); if (ms.getLastRow() > 1) { const rows = ms.getRange(2, 1, ms.getLastRow() - 1, MONTHLY_COLS.length).getValues(); rows.forEach(r => { const key = normalizeKey(r[0]); if (key) result.monthly[key] = rowToMonthly(r); }); } // Memos const mms = SS.getSheetByName(SHEETS.MEMOS); if (mms.getLastRow() > 1) { const rows = mms.getRange(2, 1, mms.getLastRow() - 1, MEMO_COLS.length).getValues(); result.memos = rows.map(r => rowToMemo(r)); } // Goals [['weeklyGoals', SHEETS.WEEKLY_GOALS], ['monthlyGoals', SHEETS.MONTHLY_GOALS]].forEach(([camel, sn]) => { const gs = SS.getSheetByName(sn); if (gs.getLastRow() > 1) { const rows = gs.getRange(2, 1, gs.getLastRow() - 1, GOAL_COLS.length).getValues(); result[camel] = rowToGoalObj(rows); } }); return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); } // ===== POST: データを保存 ===== function doPost(e) { initSheets(); const payload = JSON.parse(e.postData.contents); const action = payload.action || 'saveAll'; if (action === 'saveAll') { const data = payload.data; // Daily if (data.daily !== undefined) { const sheet = SS.getSheetByName(SHEETS.DAILY); if (sheet.getLastRow() > 1) sheet.deleteRows(2, sheet.getLastRow() - 1); Object.entries(data.daily || {}).forEach(([key, val]) => { const row = sheet.getLastRow() + 1; sheet.appendRow(dailyToRow(key, val)); writeKeyCell(sheet, row, 1, key); }); } // Weekly if (data.weekly !== undefined) { const sheet = SS.getSheetByName(SHEETS.WEEKLY); if (sheet.getLastRow() > 1) sheet.deleteRows(2, sheet.getLastRow() - 1); Object.entries(data.weekly || {}).forEach(([key, val]) => { const row = sheet.getLastRow() + 1; sheet.appendRow(weeklyToRow(key, val)); writeKeyCell(sheet, row, 1, key); }); } // Monthly if (data.monthly !== undefined) { const sheet = SS.getSheetByName(SHEETS.MONTHLY); if (sheet.getLastRow() > 1) sheet.deleteRows(2, sheet.getLastRow() - 1); Object.entries(data.monthly || {}).forEach(([key, val]) => { const row = sheet.getLastRow() + 1; sheet.appendRow(monthlyToRow(key, val)); writeKeyCell(sheet, row, 1, key); }); } // Memos if (data.memos !== undefined) { const sheet = SS.getSheetByName(SHEETS.MEMOS); if (sheet.getLastRow() > 1) sheet.deleteRows(2, sheet.getLastRow() - 1); (data.memos || []).forEach(memo => { sheet.appendRow(memoToRow(memo)); }); } // Goals [['weeklyGoals', SHEETS.WEEKLY_GOALS], ['monthlyGoals', SHEETS.MONTHLY_GOALS]].forEach(([camel, sn]) => { if (data[camel] !== undefined) { const sheet = SS.getSheetByName(sn); if (sheet.getLastRow() > 1) sheet.deleteRows(2, sheet.getLastRow() - 1); Object.entries(data[camel] || {}).forEach(([key, val]) => { const row = sheet.getLastRow() + 1; sheet.appendRow(goalToRow(key, val)); writeKeyCell(sheet, row, 1, key); }); } }); } return ContentService.createTextOutput(JSON.stringify({ status: 'ok' })).setMimeType(ContentService.MimeType.JSON); }
